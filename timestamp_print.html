This utility function prints the current timestamp with milliseconds using <chrono>, <iomanip>, and <ctime>. Useful for debugging or logging in time-sensitive systems like embedded or multithreaded applications.

#include <chrono>
#include <iomanip>
#include <ctime>
#include <cstring>  // for strlen

// Helper: Print current timestamp
void print_timestamp(const std::string& label) {
    using namespace std::chrono;
    auto now = system_clock::now();
    auto now_time_t = system_clock::to_time_t(now);
    auto now_ms = duration_cast<milliseconds>(now.time_since_epoch()) % 1000;

    std::tm now_tm = *std::localtime(&now_time_t);
    std::cout << "[" << label << "] "
              << std::put_time(&now_tm, "%Y-%m-%d %H:%M:%S")
              << "." << std::setfill('0') << std::setw(3) << now_ms.count()
              << std::endl;
}

// Calling function
print_timestamp("triggered write callback");
